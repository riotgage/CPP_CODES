#include<iostream>
using namespace std;
int main(){
    int input;
    cin>>input;
    cout<<input;
    while(input--){
        char ch;
        cin>>ch;
        string str;
        cin>>str;
        int len=str.size();
        if(len%2==1){
            str[len-1]='a'+'z'-str[len-1];
        }
    
        for(int i=0;i<len-1;i=i+2){
            str[i+1]='a'+'z'-str[i+1];
            str[i]='a'+'z'-str[i];
            ch=str[i];
            str[i]=str[i+1];
            str[i+1]=ch;
        }
        cout<<str<<endl;
    }
    
}
